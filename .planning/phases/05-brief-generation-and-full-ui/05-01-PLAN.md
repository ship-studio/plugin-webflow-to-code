    **Step 2: Implement src/brief/io.ts**

    Follow the exact pattern from the Figma plugin's `plugin-figma/src/brief/io.ts`. The Shell type is the same interface used throughout this plugin — import it from context or define inline as `{ exec(cmd: string, args: string[]): Promise<{ exit_code: number; stdout: string; stderr: string }> }`.

    ```typescript
    export async function saveBrief(
      shell: { exec(cmd: string, args: string[]): Promise<{ exit_code: number; stdout: string; stderr: string }> },
      projectPath: string,
      markdown: string,
    ): Promise<void> {
      const briefPath = `${projectPath}/.shipstudio/assets/brief.md`;
      const encoded = btoa(unescape(encodeURIComponent(markdown)));
      const result = await shell.exec('bash', ['-c', `echo '${encoded}' | base64 -d > '${briefPath}'`]);
      if (result.exit_code !== 0) {
        throw new Error(`Failed to save brief: ${result.stderr}`);
      }
    }

    export async function copyToClipboard(
      shell: { exec(cmd: string, args: string[]): Promise<{ exit_code: number; stdout: string; stderr: string }> },
      markdown: string,
    ): Promise<void> {
      const encoded = btoa(unescape(encodeURIComponent(markdown)));
      const result = await shell.exec('bash', ['-c', `echo '${encoded}' | base64 -d | pbcopy`]);
      if (result.exit_code !== 0) {
        throw new Error(`Clipboard copy failed: ${result.stderr}`);
      }
    }
    ```
